#include <Servo.h> 
const int trigPin = 9; 
const int echoPin = 10; 
const int servoPin = 6; 
Servo doorServo; 
long duration; 
int distance; 
void setup() { 
Serial.begin(9600); 
pinMode(trigPin, OUTPUT); 
pinMode(echoPin, INPUT); 
doorServo.attach(servoPin); 
doorServo.write(0);  // Initially door closed 
} 
void loop() { 
// Clear the trigPin 
digitalWrite(trigPin, LOW); 
delayMicroseconds(2); 
// Trigger the sensor by setting trigPin high for 10 microseconds 
digitalWrite(trigPin, HIGH); 
delayMicroseconds(10); 
digitalWrite(trigPin, LOW); 
// Read the echoPin, returns the sound wave travel time in microseconds 
duration = pulseIn(echoPin, HIGH); 
// Calculate the distance in cm 
distance = duration * 0.034 / 2; 
Serial.print("Distance: "); 
Serial.print(distance); 
Serial.println(" cm"); 
if (distance > 0 && distance < 20) { 
// Object detected close, open door 
doorServo.write(90);  // Adjust angle to open door as needed 
} else { 
// No object close, close door 
doorServo.write(0);  // Door closed position 
} 
delay(500);  // Wait before next reading 
}
