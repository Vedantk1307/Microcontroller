import RPi.GPIO as GPIO 
import time 
 
# Set GPIO pin numbering mode 
GPIO.setmode(GPIO.BCM) 
 
# Define GPIO pins for 4 LEDs 
led_pins = [17, 18, 27, 22] 
 
# Set all pins as output 
for pin in led_pins: 
    GPIO.setup(pin, GPIO.OUT) 
 
print("4 LED Bank Interfacing with Raspberry Pi") 
print("Press Ctrl+C to stop the program") 
 
try: 
    while True: 
        # Sequential ON 
        for pin in led_pins: 
            GPIO.output(pin, GPIO.HIGH) 
            time.sleep(0.5) 
         
        # Sequential OFF 
        for pin in led_pins: 
            GPIO.output(pin, GPIO.LOW) 
            time.sleep(0.5) 
         
        # Blink all LEDs together 
        print("Blinking all LEDs...") 
        for i in range(3): 
            for pin in led_pins: 
                GPIO.output(pin, GPIO.HIGH) 
            time.sleep(0.5) 
            for pin in led_pins: 
                GPIO.output(pin, GPIO.LOW) 
            time.sleep(0.5) 
         
        # Chasing pattern 
        print("Running LED pattern...") 
        for i in range(4): 
            GPIO.output(led_pins[i], GPIO.HIGH) 
            time.sleep(0.3) 
            GPIO.output(led_pins[i], GPIO.LOW) 
 
except KeyboardInterrupt: 
    print("\nProgram stopped by user") 
finally: 
    GPIO.cleanup() 
    print("GPIO cleanup done.")
